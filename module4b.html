Code Node — Run Once for All Items (loop) 
const results = [];

for (const item of $input.all()) {
  const raw =
    item.json?.message?.content?.[0]?.text
    ?? item.json?.choices?.[0]?.message?.content
    ?? item.json?.candidates?.[0]?.content?.parts?.[0]?.text ?? '';

  const clean = raw.replace(/^```json\s*/i, '').replace(/```\s*$/i, '').trim();

  try {
    results.push({ json: JSON.parse(clean) });
  } catch (e) {
    // fallback ถ้า parse ไม่ได้
    results.push({ json: { severity: 'medium', key_issue: raw.slice(0, 200), parse_error: e.message } });
  }
}

return results;
